// ================= 3D EFECTO =================
const foto = document.getElementById("foto3d");

foto.addEventListener("mousemove", (e) => {
  const rect = foto.getBoundingClientRect();
  const x = e.clientX - rect.left;
  const y = e.clientY - rect.top;

  const midX = rect.width / 2;
  const midY = rect.height / 2;

  const rotateX = ((y - midY) / midY) * 10;
  const rotateY = ((x - midX) / midX) * 10;

  foto.style.transform = `rotateX(${-rotateX}deg) rotateY(${rotateY}deg) scale(1.05)`;
});

foto.addEventListener("mouseleave", () => {
  foto.style.transform = "rotateX(0) rotateY(0)";
});

function moverFoto(clientX,clientY){

    const rect=foto.getBoundingClientRect();

    const x=clientX-rect.left;
    const y=clientY-rect.top;

    const midX=rect.width/2;
    const midY=rect.height/2;

    const rotateX=((y-midY)/midY)*10;
    const rotateY=((x-midX)/midX)*10;

    foto.style.transform=
    `rotateX(${-rotateX}deg)
     rotateY(${rotateY}deg)
     scale(1.05)`;

}

foto.addEventListener("touchmove",function(e){

    moverFoto(
        e.touches[0].clientX,
        e.touches[0].clientY
    );

});

foto.addEventListener("touchend",function(){

    foto.style.transform="rotateX(0) rotateY(0)";

});

// ================= FONDO =================
const canvas = document.getElementById("bg");
const ctx = canvas.getContext("2d");

canvas.width = window.innerWidth;
canvas.height = window.innerHeight;

let particles = [];
for (let i = 0; i < 60; i++) {
  particles.push({
    x: Math.random()*canvas.width,
    y: Math.random()*canvas.height,
    dx: Math.random()-0.5,
    dy: Math.random()-0.5
  });
}

function animate() {
  ctx.clearRect(0,0,canvas.width,canvas.height);
  particles.forEach(p=>{
    p.x+=p.dx;
    p.y+=p.dy;
    ctx.beginPath();
    ctx.arc(p.x,p.y,2,0,Math.PI*2);
    ctx.fillStyle="#38bdf8";
    ctx.fill();
  });
  requestAnimationFrame(animate);
}
animate();

// ================= TYPING =================
const roles = ["Analista de Redes","Automatización de Procesos","5S, Kaizen y Six Sigma "];
let i=0,c=0,d=false;

function type(){
  const el=document.querySelector(".typing");
  const t=roles[i];

  if(!d){
    c++;
    el.textContent=t.substring(0,c);
    if(c===t.length){ d=true; setTimeout(type,1500); return;}
  } else {
    c--;
    el.textContent=t.substring(0,c);
    if(c===0){ d=false; i=(i+1)%roles.length;}
  }

  setTimeout(type,d?60:120);
}
type();
